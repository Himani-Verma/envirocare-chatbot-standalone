4. WordPress Integration

Add to your theme's functions.php or use a plugin:

// Add to functions.php
function add_envirocare_chatbot() {
    echo '<div id="envirocare-chatbot">
        <iframe 
            src="https://your-domain.com/chatbot.html" 
            width="400" 
            height="600" 
            frameborder="0"
            title="Envirocare Labs Chatbot"
            style="border-radius: 15px; box-shadow: 0 4px 20px rgba(0,0,0,0.1);"
        ></iframe>
    </div>';
}
add_action('wp_footer', 'add_envirocare_chatbot');

// Or add to specific pages only
function add_chatbot_to_contact_page() {
    if (is_page('contact')) {
        add_envirocare_chatbot();
    }
}
add_action('wp_footer', 'add_chatbot_to_contact_page');
